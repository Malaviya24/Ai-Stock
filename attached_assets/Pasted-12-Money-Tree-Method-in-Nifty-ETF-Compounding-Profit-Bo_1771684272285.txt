# 🧮 MONEY TREE ENGINE

```javascript
export function moneyTreeEngine({
  price,
  totalUnits,
  totalPrincipal,
  weeklyInvestment,
}) {

  const threshold = weeklyInvestment * 0.20;

  const portfolioValue = totalUnits * price;
  const profit = portfolioValue - totalPrincipal;

  let sellUnits = 0;
  let bookedProfit = 0;

  if (profit >= threshold) {
    sellUnits = threshold / price;
    bookedProfit = threshold;
    totalUnits -= sellUnits;
  }

  return {
    totalUnits,
    portfolioValue,
    profit,
    sellUnits,
    bookedProfit
  };
}
```
