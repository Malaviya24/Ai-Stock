# 🧮 CALCULATE 55 DAY HIGH

```javascript
function calculate55DayHigh(data, index) {
  const start = Math.max(0, index - 54);
  const slice = data.slice(start, index + 1);
  return Math.max(...slice.map(d => d.high));
}
```

# 🐢 TURTLE ENGINE

```javascript
export function turtleEngine(data, capital = 20000) {

  const perOrder = capital / 15;

  let units = 0;
  let invested = 0;
  let entryCount = 0;
  let trades = [];

  for (let i = 55; i < data.length; i++) {

    const today = data[i];
    const high55 = calculate55DayHigh(data, i);

    if (today.close >= high55 && entryCount < 15) {

      const shares = perOrder / today.close;
      units += shares;
      invested += perOrder;
      entryCount++;

      trades.push({
        type: "BUY",
        date: today.date,
        price: today.close
      });
    }

    if (units > 0) {

      const avgPrice = invested / units;

      if (today.close >= avgPrice * 1.0628) {

        trades.push({
          type: "SELL",
          date: today.date,
          price: today.close
        });

        units = 0;
        invested = 0;
        entryCount = 0;
      }
    }
  }

  return trades;
}
```
