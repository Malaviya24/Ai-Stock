## 2️⃣ API Route Example

`/app/api/reit-scan/route.js`

```javascript
import yahooFinance from "yahoo-finance2";

const symbols = [
  "EMBASSY.NS",
  "MINDSPACE.NS",
  "BIRET.NS",
  "INDIGRID.NS",
  "PGINVIT.NS"
];

export async function GET() {
  const results = [];

  for (let symbol of symbols) {
    const data = await yahooFinance.historical(symbol, {
      period1: "2023-01-01",
      interval: "1d"
    });

    const last5 = data.slice(-6, -1);
    const today = data[data.length - 1];

    const highestHigh = Math.max(...last5.map(d => d.high));
    const avgVolume = last5.reduce((a, b) => a + b.volume, 0) / 5;

    const close = today.close;
    const volume = today.volume;

    const buySignal =
      close > highestHigh &&
      volume > 1.5 * avgVolume;

    results.push({
      symbol,
      buySignal,
      entry: close,
      target: close * 1.06,
      stop: close * 0.96
    });
  }

  return Response.json(results);
}
```
