# 📊 STRATEGY ENGINE

```javascript
export function gapUpEngine({
  previousClose,
  open,
  currentPrice,
  entryPrice,
  units,
  invested
}) {

  const gapPercent = calculateGap(previousClose, open);

  let signal = "NO TRADE";

  if (gapPercent >= 3.14) {

    if (currentPrice > open) {
      signal = "BUY";
    }
  }

  let target = 0;

  if (entryPrice) {
    target = entryPrice * 1.0628;

    if (currentPrice >= target) {
      signal = "SELL";
    }
  }

  return {
    gapPercent,
    signal,
    target
  };
}
```
